<script>

  /* ── Helpers ── */

  const $ = id => document.getElementById(id);


  function utcHour() {

    const d = new Date();

    return d.getUTCHours() + d.getUTCMinutes() / 60;

  }


  function utcDayOfWeek() {

    return new Date().getUTCDay(); // 0=Sun, 6=Sat

  }


  // Peak: 13:00–19:00 UTC on weekdays

  // Transitional: 12:00–13:00 and 19:00–20:00

  function getLoad(h, isWeekend) {

    if (isWeekend) return 'off';

    if (h >= 13 && h < 19) return 'peak';

    if ((h >= 12 && h < 13) || (h >= 19 && h < 20)) return 'transitional';

    return 'off';

  }


  /* ── Live clock ── */

  function updateClock() {

    const now = new Date();

    const utc = now.toUTCString().replace('GMT', 'UTC');

    const local = now.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit', second: '2-digit' });

    $('live-clock').innerHTML =

      `<div>${local} local</div>` +

      `<div style="color:var(--ink-faint)">${now.getUTCHours().toString().padStart(2,'0')}:${now.getUTCMinutes().toString().padStart(2,'0')} UTC</div>` +

      `<div style="font-size:0.65rem;margin-top:2px;color:var(--ink-faint)">Updates live</div>`;

  }


  /* ── Status Banner ── */

  function updateStatus() {

    const h = utcHour();

    const dow = utcDayOfWeek();

    const isWeekend = dow === 0 || dow === 6;

    const load = getLoad(h, isWeekend);


    const banner = $('status-banner');

    const dot = $('status-dot');


    const colors = {

      peak:        'var(--rust)',

      transitional:'var(--gold)',

      off:         'var(--sage)'

    };


    const color = colors[load];

    banner.style.setProperty('--status-color', color);

    dot.style.setProperty('--status-color', color);

      },

      transitional: {

        label: '🟡 Transitional period',

        detail: 'Load is shifting. You may experience some throttling, but it's not full peak. Best to monitor your session progress.',

      },

        label: isWeekend ? '🟢 Weekend — fully off-peak' : '🟢 Off-peak — full capacity',

        detail: isWeekend

          ? 'Weekends are entirely unrestricted. A great time for long, intensive sessions.'

          : 'Session limits are at their most generous right now. Ideal for long or token-intensive work.',

        meta:   isWeekend ? 'Off-peak all day Sat & Sun' : `Next peak: ${h < 13 ? '13:00 UTC today' : '13:00 UTC tomorrow (if weekday)'}`

      }

    };


    const cfg = configs[load];

    $('status-label').textContent = cfg.label;

    $('status-detail').textContent = cfg.detail;

    $('status-meta').textContent   = cfg.meta;


    // Update estimates

    updateEstimates(load);

  }


  /* ── Estimates ── */

  function updateEstimates(load) {

    // Token budgets: Pro ~44k, Max5 ~88k, Max20 ~220k

    // Avg conversational exchange ~650 tokens

    // During peak, budget drains ~1.4–2x faster (we model 1.6x)

    const multiplier = load === 'peak' ? 1 / 1.6 : 1;

    const plans = [

      { name: 'Free / Pro', tokens: 44000,  price: '$0–$20/mo' },

      { name: 'Max 5×',     tokens: 88000,  price: '$100/mo'   },

      { name: 'Max 20×',    tokens: 220000, price: '$200/mo'   },

    ];


    const avgTokens = 650; // per exchange (input+output)


    const html = plans.map(p => {

      const msgs = Math.round((p.tokens * multiplier) / avgTokens);

      const isPeak = load === 'peak';

      return `

        <div class="estimate-card">

          <div class="estimate-plan">${p.name}</div>

          <div class="estimate-num">~${msgs}<span> msgs</span></div>

          <div class="estimate-desc">per 5-hr session<br>${p.price}</div>

          ${isPeak ? `<div class="estimate-peak-note">↓ reduced by ~40% at peak</div>` : '<div class="estimate-peak-note" style="color:var(--sage)">✓ full capacity</div>'}

        </div>`;

    }).join('');


    $('estimates-grid').innerHTML = html;

  }


  /* ── Timeline marker ── */

  function updateMarker() {

    const h = utcHour();

    const pct = (h / 24) * 100;

    $('time-marker').style.left = `${pct}%`;

  }


  /* ── Weekly grid ── */

  function buildWeekly() {

    const days = ['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'];

    const today = utcDayOfWeek();

    const notes = {

      0: { label:'off', note:'Fully off-peak',     bg:'var(--sage)', text:'var(--cream)' },

      1: { label:'pkd', note:'Heavy peak AM (US)',  bg:'var(--rust)', text:'var(--cream)' },

      2: { label:'pkd', note:'Peak 13–19 UTC',      bg:'var(--rust)', text:'var(--cream)' },

      3: { label:'pkd', note:'Peak 13–19 UTC',      bg:'var(--rust)', text:'var(--cream)' },

      4: { label:'pkd', note:'Peak 13–19 UTC',      bg:'var(--rust)', text:'var(--cream)' },

      5: { label:'pkd', note:'Peak 13–19 UTC',      bg:'var(--rust)', text:'var(--cream)' },

      6: { label:'off', note:'Fully off-peak',      bg:'var(--sage)', text:'var(--cream)' },

    };

    const html = days.map((d, i) => {

      const n = notes[i];

      const isToday = i === today;

      const isWknd  = i === 0 || i === 6;

      return `

        <div class="day-card${isToday ? ' today' : ''}${isWknd ? ' weekend' : ''}">

          <div class="day-name">${d}</div>

          <div class="day-indicator" style="background:${n.bg};color:${n.text};font-size:0.55rem">

            ${isToday ? 'now' : n.label}

          </div>

          <div class="day-note">${n.note}</div>

        </div>`;

    }).join('');

    $('day-grid').innerHTML = html;

  }


  /* ── Init & tick ── */

  function tick() {

    updateClock();

    updateStatus();

    updateMarker();

  }


  buildWeekly();

  tick();

  setInterval(tick, 10000);        // update every 10 s

  setInterval(buildWeekly, 60000); // rebuild day grid every minute (handles day rollover)
